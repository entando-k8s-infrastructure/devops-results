package org.entando.entando.ent.util;

/**
 * Santization library
 * <p>
 * Set of classes to help logging in Entando.
 * <p>
 * The classes supports simple sanitization in a scanner-friendly way.
 */
public interface EntSanitization {

	interface EntSanitizable<T> {
		T santize();
	}

	class EntSanitizedString implements EntSanitizable<String> {

		public final String value;

		EntSanitizedString(String value) {
			this.value = value;
		}

		@Override
		public String toString() {
			return santize();
		}

		@Override
		public String santize() {
			return fixJavaSecS5145(this.value);
		}
	}

	/**
	 * EntSanitizedString factory method
	 */
	static EntSanitizedString sanitize(String stringValue) {
		return new EntSanitizedString(stringValue);
	}

	static String fixJavaSecS5145(String s) {
		return (s == null) ? null : s.replace("\n", "_").replace("\r", "_").replace("\t", "_");
	}
}
